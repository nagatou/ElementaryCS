# junkan.py
# 配列の使い方の練習（循環小数を循環するまで求める）
# 入力: d
# 出力: 1/d の各桁を循環するまで求める
d = int(input("1/d d(>=2)? "))
print("1/",d," を求めます")
x = 1
print("0.", end=""),
while (True):
  x = x * 10
  q = x // d
  x = x % d
  print(q, end="")
  if x == 0:
    break
print("")
