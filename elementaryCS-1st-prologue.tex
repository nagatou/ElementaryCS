\subsection{評価基準}
\begin{frame}
\frametitle{評価基準}
  \begin{itemize}
\item 講義は全 7 回，期末試験は行いません
\item 宿題: 3 回 くらい（提出不要）
\item 課題: 4 回 \(25+30+30=85\) 点
\item 特別課題: 1 回 \(15\) 点（提出任意）
\item 宿題・課題提出: 
    \begin{itemize}
\item 講義時間中に課題を出します
\item 提出方法はその都度指定します
    \end{itemize}
\item 宿題と課題提出で出欠確認に変えます
  \end{itemize}
\end{frame}
